word_cloud_corpus <-
    sort(table(word_cloud_corpus$媒介物), decreasing = TRUE)
word_cloud_corpus <- as.data.frame(word_cloud_corpus)
colnames(word_cloud_corpus) <- c("媒介物", "頻率")
